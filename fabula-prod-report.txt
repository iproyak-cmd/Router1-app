def complete_paid_site_order(order: sqlite3.Row, payload: dict[str, Any], external_id: str | None) -> dict[str, Any]:
    if order["status"] == "paid" and order["conf_path"] and order["qr_path"]:
        if order["product"] in {"iphone", "iphone_test"}:
            schedule_iphone_delivery(str(order["id"]))
        return {"ok": True, "already_paid": True, "order_id": order["id"]}

    product = PRODUCTS.get(order["product"])
    if not product:
        raise RuntimeError(f"unknown product for order {order['id']}: {order['product']}")

    result = mark_payment_paid_and_activate(
        payment_id=int(order["internal_payment_id"]),
        external_payment_id=external_id,
        raw_payload=json.dumps(payload, ensure_ascii=False),
    )
    if not result.get("ok"):
        raise RuntimeError(f"payment activation failed: {result}")

    generation: dict[str, Any] | None = None
    conf_path = order["conf_path"]
    qr_path = order["qr_path"]
    device_id = result.get("device_id") or order["device_id"]

    if device_id and result.get("does_create_config"):
        relabel_generated_device(int(device_id), product, order["product"])
        if result.get("reused_test_device"):
            # Апгрейд с бесплатного теста на полный тариф: mark_payment_paid_and_activate
            # уже переиспользовала существующий пир/ключи на этом device_id — конфиг не перевыпускаем,
            # клиенту не придётся заново настраивать роутер/устройство.
            with db() as conn:
                existing_row = conn.execute(
                    "SELECT config_path, conf_path, qr_path FROM client_configs WHERE id = ?",
                    (int(device_id),),
                ).fetchone()
            if existing_row:
                conf_path = existing_row["config_path"] or existing_row["conf_path"] or conf_path
                qr_path = existing_row["qr_path"] or qr_path
        elif not result.get("already_paid") or not conf_path:
            generation = generate_for_client_device(int(result["client_id"]), int(device_id))
            if generation.get("ok"):
                conf_path = str(generation.get("config_path") or "")
                qr_path = str(generation.get("qr_path") or "")
            else:
                logger.error("WG generation failed for order %s: %s", order["id"], generation)

    if product.get("requires_setup_request") and not result.get("already_paid"):
        create_setup_request(
            int(result["client_id"]),
            "router",
            "site_order",
            comment=f"Заказ с сайта {order['id']}",
        )

    paid_at = to_iso(utc_now())
    with db() as conn:
        conn.execute(
            """
            UPDATE site_orders
            SET status = 'paid',
                paid_at = COALESCE(paid_at, ?),
                device_id = COALESCE(?, device_id),
                conf_path = COALESCE(?, conf_path),
                qr_path = COALESCE(?, qr_path),
                generation_status = ?,
                generation_error = ?,
                raw_payload = ?,
                updated_at = ?
            WHERE id = ?
            """,
            (
                paid_at,
                int(device_id) if device_id else None,
                conf_path,
                qr_path,
                "ready" if conf_path and qr_path else "generation_error",
                None if conf_path and qr_path else json.dumps(generation or {}, ensure_ascii=False),
                json.dumps(payload, ensure_ascii=False),
                paid_at,
                order["id"],
            ),
        )
        conn.commit()
    notify_payment_success(
        payment_id=int(order["internal_payment_id"]) if order["internal_payment_id"] else None,
        client_id=int(result["client_id"]) if result.get("client_id") else int(order["client_id"]),
        source="site",
        device_id=int(device_id) if device_id else None,
        order_id=str(order["id"]),
        generation=generation,
    )
    if order["product"] in {"iphone", "iphone_test"} and conf_path:
        schedule_iphone_delivery(str(order["id"]))
    return {"ok": True, "order_id": order["id"], "generation": generation}

def _complete_free_trial_order(
    order_id: str,
    marker: dict[str, Any],
    client_id: int,
    payment_id: int,
) -> None:
    try:
        with db() as conn:
            order = conn.execute("SELECT * FROM site_orders WHERE id = ?", (order_id,)).fetchone()
        if order is None:
            raise RuntimeError("free trial order not found")
        activation = complete_paid_site_order(
            order,
            marker,
            f"free-trial-{client_id}-{payment_id}",
        )
        generation = activation.get("generation") if isinstance(activation, dict) else None
        status = "generation_error" if isinstance(generation, dict) and not generation.get("ok") else "active"
        config_id = None
        if isinstance(activation, dict):
            raw_config_id = activation.get("device_id") or activation.get("config_id")
            if raw_config_id:
                config_id = int(raw_config_id)
        if isinstance(generation, dict):
            raw_config_id = generation.get("device_id") or generation.get("config_id")
            if raw_config_id:
                config_id = int(raw_config_id)
        if marker.get("source") == "fabula_free_access":
            if config_id:
                # Fabula uses a stable direct full tunnel. The optional
                # Router1 AI transparent redirect can leave Android with a
                # handshake but no usable DNS/Telegram/YouTube traffic.
                from r1pro_admin import use_direct_routing_for_device

                use_direct_routing_for_device(client_id, config_id)
            with db() as conn:
                conn.execute(
                    "UPDATE fabula_access_claims SET status = ?, config_id = COALESCE(?, config_id), updated_at = ? WHERE client_id = ? AND device_type = ?",
                    (status, config_id, to_iso(utc_now()), client_id, marker.get("product")),
                )
                conn.commit()
        else:
            update_free_trial_claim(client_id, status, payment_id)
        logger.info(
            "free_trial_completed client_id=%s order_id=%s payment_id=%s status=%s",
            client_id,
            order_id,
            payment_id,
            status,
        )
    except Exception as exc:
        if marker.get("source") == "fabula_free_access":
            with db() as claim_conn:
                claim_conn.execute(
                    "UPDATE fabula_access_claims SET status = 'activation_error', updated_at = ? WHERE client_id = ? AND device_type = ?",
                    (to_iso(utc_now()), client_id, marker.get("product")),
                )
                claim_conn.commit()
        else:
            update_free_trial_claim(client_id, "activation_error", payment_id)
        with db() as conn:
            conn.execute(
                "UPDATE site_orders SET status = 'generation_error', generation_error = ?, updated_at = ? WHERE id = ?",
                (str(exc)[:500], to_iso(utc_now()), order_id),
            )
            conn.commit()
        logger.exception("free_trial_activation_failed client_id=%s order_id=%s", client_id, order_id)

def _app_billing_offers_for_client(client_id: int) -> list[dict[str, Any]]:
    with connect() as conn:
        rows = conn.execute(
            """
            SELECT * FROM client_configs
            WHERE client_id=? AND COALESCE(config_path, conf_path, '') != ''
            ORDER BY id DESC
            """,
            (client_id,),
        ).fetchall()
    has_full_purchase = any(
        (row["subscription_type"] or "").lower() != "test"
        and (row["payment_status"] or "").lower() in {"paid", "active", "manual_grant"}
        for row in rows
    )
    if has_full_purchase:
        return APP_RENEWAL_OFFERS
    test_rows = [row for row in rows if (row["subscription_type"] or "").lower() == "test"]
    if not test_rows:
        return []
    row = test_rows[0]
    raw = " ".join(str(row[key] or "") for key in ("product_type", "device_type", "device_name")).lower()
    if "router" in raw or "роут" in raw:
        key = "router_specialist"
    elif "iphone" in raw or "ios" in raw:
        key = "iphone_wg"
    elif "pc" in raw or "laptop" in raw or "ноут" in raw or "комп" in raw:
        key = "pc_xray"
    else:
        key = "smartphone_wg"
    return [APP_TRIAL_CONVERSION_OFFERS[key]]

async def _app_renewal_order(request: Request) -> dict[str, Any]:
    _require_app_auth(request)
    try:
        body = await request.json()
    except Exception:
        raise HTTPException(status_code=400, detail="invalid json")
    phone = str(body.get("phone") or "").strip()
    offer_key = str(body.get("offer_key") or "").strip()
    if offer_key not in APP_RENEWAL_KEYS | set(APP_TRIAL_CONVERSION_OFFERS):
        raise HTTPException(status_code=400, detail="unknown renewal offer")
    with connect() as conn:
        client = _app_client_by_identity(conn, phone, None)
    if client is None:
        raise HTTPException(status_code=404, detail="client not found")
    allowed = {item["key"] for item in _app_billing_offers_for_client(int(client["id"]))}
    if offer_key not in allowed:
        raise HTTPException(status_code=409, detail="offer is not available for this client")
    result = create_billing_payment_draft(int(client["id"]), offer_key)
    if not result.get("ok") or not result.get("payment_url"):
        logger.error("app_renewal_payment_failed client_id=%s offer=%s result=%s", client["id"], offer_key, result)
        raise HTTPException(status_code=503, detail="payment temporarily unavailable")
    return {
        "ok": True,
        "payment_id": result.get("payment_id"),
        "amount": result.get("amount"),
        "payment_url": result.get("payment_url"),
        "text": result.get("text"),
    }
